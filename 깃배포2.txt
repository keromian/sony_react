create table user(
    userId varchar(30) not null primary key comment '아이디',
    userPw varchar(16) not null comment '비밀번호',
    userName varchar(30) not null comment '이름',
    userEmail varchar(250) null comment '이메일' 
)

create table bbs(
    bbsId int not null primary key comment '번호',
    userId varchar(30) not null comment '아이디',
    subject varchar(1000) not null comment '글제목',
    content varchar(4096) not null comment '글내용',
    writeDate varchar(20) comment '작성일',
    deleteOk int comment '삭제여부',
    hit int comment '조회수' 
)
